#!/usr/bin/env bash
if [ $# -eq 1 ]; then
    if [ $1 = "start" ]; then
        uwsgi --ini etc/uwsgi.ini
    elif [ $1 = "reload" ]; then
        uwsgi --reload .uwsgi.pid
    elif [ $1 = "restart" ]; then
        uwsgi --stop .uwsgi.pid
        sleep 3
        uwsgi --ini etc/uwsgi.ini
    elif [ $1 = "stop" ]; then
        uwsgi --stop .uwsgi.pid
    fi
else
    echo "Usage: uwsgictl <start | reload | stop>"
fi
